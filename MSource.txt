# --- Step2: Houdini UI の一部を隠す検証コード ---

import hou
from PySide2 import QtWidgets

def list_ui_objects():
    """デバッグ用：Houdini メインウィンドウの子 Widgets を一覧表示"""
    mw = hou.qt.mainWindow()
    for child in mw.findChildren(QtWidgets.QWidget):
        print(child.objectName(), " : ", type(child))

def hide_ui_elements():
    """判明している UI の一部を隠す"""
    mw = hou.qt.mainWindow()

    for child in mw.findChildren(QtWidgets.QWidget):
        child.hide()

    mw.resize(200, 200)

def open_scene_and_parameters():
    """シーンとパラメーターを表示"""
    # 現在のデスクトップを取得
    desktop = hou.ui.curDesktop()
    
    # Scene Viewer フローティングタブを作成
    scene_tab = desktop.createFloatingPaneTab(
        hou.paneTabType.SceneViewer,
        position=(100, 100),  # 任意の位置
        size=(800, 600),      # 任意のサイズ
        python_panel_interface=None,
        immediate=True
    )
    
    # Parameters フローティングタブを作成
    parm_tab = desktop.createFloatingPaneTab(
        hou.paneTabType.Parm,
        position=(950, 100),  # Scene Viewer の右側に配置
        size=(300, 600),      # 任意のサイズ
        python_panel_interface=None,
        immediate=True
    )


#### m a i n #####################################
hide_ui_elements()

open_scene_and_parameters()
