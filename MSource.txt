// ============================================================
// Node
// ============================================================

struct Node
{
    std::string name;
    std::string path;

    bool hasTransform = false;

    pxr::GfMatrix4d transform = pxr::GfMatrix4d(1.0);
    pxr::GfMatrix4d worldTransform = pxr::GfMatrix4d(1.0);

    int parent = -1;

    std::vector<int> children;
    std::vector<int> staticMeshes;
};

// ============================================================
// Mesh
// ============================================================

struct Mesh
{
    std::string name;

    pxr::VtIntArray   faces;
    pxr::VtIntArray   indices;
    pxr::VtVec3fArray points;
};

// ============================================================
// exporter
// ============================================================

class exporter
{
public:
    bool main(const pxr::SdfLayer& layer);

private:
    // ----------------------------
    // Core Data
    // ----------------------------
    std::vector<int>  rootNodes;
    std::vector<Node> nodes;
    std::vector<Mesh> meshes;

    // ----------------------------
    // Read
    // ----------------------------
    bool readPrim(const pxr::UsdPrim& prim, int parent);
    bool readXform(const pxr::UsdPrim& prim, int parent);
    bool readMesh(const pxr::UsdPrim& prim, int parent);

    std::pair<int, Node&> addNode(int parent);

    void readXformInternal(Node& node,
        const pxr::UsdPrim& prim,
        int parent);

    void readTransform(const pxr::UsdPrim& prim,
        Node& node,
        int parent);

    // ----------------------------
    // Export 
    // ----------------------------
    void exportData();
    void writeNode(size_t nodeIndex,
        const pxr::GfMatrix4d& correctionTransform);

    void exportMesh(const pxr::GfMatrix4d& worldTransform,
        int nodeIndex,
        int meshIndex);
};

■CPP
#include "exporter.h"

PXR_NAMESPACE_OPEN_SCOPE

// ============================================================
// main
// ============================================================

bool exporter::main(const SdfLayer& constLayer)
{
    auto layer = SdfCreateNonConstHandle<SdfLayer>(&constLayer);
    UsdStageRefPtr stage = UsdStage::Open(layer);

    if (!stage)
        return false;

    UsdPrim defaultPrim;

    if (stage->HasDefaultPrim())
        defaultPrim = stage->GetDefaultPrim();

    if (defaultPrim)
    {
        readPrim(defaultPrim, -1);
    }
    else
    {
        for (const UsdPrim& prim : stage->GetPseudoRoot().GetChildren())
        {
            readPrim(prim, -1);
        }
    }

    exportData();

    return true;
}

// ============================================================
// readPrim
// ============================================================

bool exporter::readPrim(const UsdPrim& prim, int parent)
{
    if (prim.IsA<UsdGeomXform>())
        return readXform(prim, parent);

    if (prim.IsA<UsdGeomMesh>())
        return readMesh(prim, parent);

    return true;
}

// ============================================================
// readXform
// ============================================================

bool exporter::readXform(const UsdPrim& prim, int parent)
{
    auto [nodeIndex, node] = addNode(parent);

    readXformInternal(node, prim, parent);

    for (const UsdPrim& child :
        prim.GetFilteredChildren(UsdTraverseInstanceProxies()))
    {
        readPrim(child, nodeIndex);
    }

    return true;
}

// ============================================================
// addNode
// ============================================================

std::pair<int, Node&> exporter::addNode(int parent)
{
    int index = static_cast<int>(nodes.size());

    nodes.emplace_back();
    Node& node = nodes.back();

    node.parent = parent;

    if (parent >= 0)
        nodes[parent].children.push_back(index);
    else
        rootNodes.push_back(index);

    return { index, node };
}

// ============================================================
// readXformInternal
// ============================================================

void exporter::readXformInternal(Node& node,
    const UsdPrim& prim,
    int parent)
{
    node.name = prim.GetName().GetString();
    node.path = prim.GetPath().GetString();

    readTransform(prim, node, parent);
}

// ============================================================
// readTransform
// ============================================================

void exporter::readTransform(const UsdPrim& prim,
    Node& node,
    int parent)
{
    UsdGeomXformable xformable(prim);

    bool reset = false;

    xformable.GetLocalTransformation(
        &node.transform,
        &reset,
        UsdTimeCode::EarliestTime());

    node.hasTransform =
        node.transform != GfMatrix4d(0.0) &&
        node.transform != GfMatrix4d(1.0);

    GfMatrix4d parentWorld =
        (parent >= 0) ? nodes[parent].worldTransform
        : GfMatrix4d(1.0);

    node.worldTransform = node.transform * parentWorld;
}

// ============================================================
// readMesh
// ============================================================

bool exporter::readMesh(const UsdPrim& prim, int parent)
{
    Mesh mesh;
    mesh.name = prim.GetName().GetString();

    UsdGeomMesh usdMesh(prim);

    usdMesh.GetFaceVertexCountsAttr().Get(&mesh.faces, 0);
    usdMesh.GetFaceVertexIndicesAttr().Get(&mesh.indices, 0);
    usdMesh.GetPointsAttr().Get(&mesh.points, 0);

    int meshIndex = static_cast<int>(meshes.size());
    meshes.push_back(mesh);

    if (parent >= 0)
        nodes[parent].staticMeshes.push_back(meshIndex);

    return true;
}

// ============================================================
// exportData
// ============================================================

void exporter::exportData()
{
    GfMatrix4d correctionTransform(1.0);

    for (int root : rootNodes)
    {
        writeNode(root, correctionTransform);
    }
}

// ============================================================
// writeNode
// ============================================================

void exporter::writeNode(size_t nodeIndex,
    const GfMatrix4d& correctionTransform)
{
    const Node& node = nodes[nodeIndex];

    GfMatrix4d finalWorld =
        node.worldTransform * correctionTransform;

    for (int meshIndex : node.staticMeshes)
    {
        exportMesh(finalWorld,
            static_cast<int>(nodeIndex),
            meshIndex);
    }

    for (int child : node.children)
    {
        writeNode(child, correctionTransform);
    }
}

// ============================================================
// exportMesh
// ============================================================

void exporter::exportMesh(const GfMatrix4d& worldTransform,
    int nodeIndex,
    int meshIndex)
{
    const Mesh& mesh = meshes[meshIndex];

    const std::string name =
        mesh.name.empty()
        ? "Node_" + std::to_string(nodeIndex) +
        "_Mesh_" + std::to_string(meshIndex)
        : mesh.name;

    // faces
    for (int face : mesh.faces)
    {
        (void)face;
    }

    // indices
    for (int index : mesh.indices)
    {
        (void)index;
    }

    // points（ワールド変換）
    for (const auto& p : mesh.points)
    {
        GfVec3f worldPos =
            GfVec3f(worldTransform.Transform(GfVec3d(p)));

        (void)worldPos;
    }
}
