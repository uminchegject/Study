// ** メッシュの作成 **
SdfPath meshPath("/World/Root/Geom/pCube1");
UsdGeomMesh mesh = UsdGeomMesh::Define(stage, meshPath);

// ** UsdSkelBindingAPI を使用してスケルトンをバインド **
UsdSkelBindingAPI bindingAPI = UsdSkelBindingAPI::Apply(mesh.GetPrim());
bindingAPI.CreateSkeletonRel().AddTarget(skeleton.GetPath());

// ** ジョイントインデックスとウェイト情報を設定 **
// 各頂点に対して影響を与えるジョイントのインデックス（ここではすべて "Root" に紐づけ）
VtArray<int> jointIndices = {0, 0, 0, 0, 0, 0, 0, 0}; // メッシュの各頂点のジョイントインデックス

// 各頂点のウェイト（ここでは "Root" の影響100%）
VtArray<float> jointWeights = {1.0f, 1.0f, 1.0f, 1.0f, 1.0f, 1.0f, 1.0f, 1.0f};

// ジョイントインデックスのPrimvarを作成
bindingAPI.CreateJointIndicesPrimvar(false, jointIndices.size()).Set(jointIndices);

// ジョイントウェイトのPrimvarを作成
bindingAPI.CreateJointWeightsPrimvar(false, jointWeights.size()).Set(jointWeights);
